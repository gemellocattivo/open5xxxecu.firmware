00034 void init_eSCI(void) {
00035 
00036 /*  Configure eSCI A                                                              */
00037 /*  Configure per PLX Devices Interface- 19200, 8 Data Bits, 1 Stop Bit, No parity*/
00038 /*  Note:  TX and Rx Pads Configured in SIU_OPS.c                                 */
00039 
00040      ESCI_A.CR2.B.BSTP = 0;          //Suppress DMA TX with errors present
00041      ESCI_A.CR2.B.MDIS = 0;          //Don't disable
00042      ESCI_A.CR2.B.RXDMA = 1;         //Activate RX DMA Channel
00043      ESCI_A.CR2.B.SBSTP = 0;          //Don't stop on bit error
00044 
00045      ESCI_A.CR1.B.SBR = CPU_CLOCK/(16*BAUD_RATE);     //SCI Baud Rate
00046      ESCI_A.CR1.B.M = 0;             //1 start bit, 8 bits, 1 stop bit
00047      ESCI_A.CR1.B.PE = 0;            //Parity Disable
00048      ESCI_A.CR1.B.PT = 0;            //Parity Type
00049      ESCI_A.CR1.B.TE = 1;            //Transmitter Enable
00050      ESCI_A.CR1.B.RE = 1;            //Receiver Enable
00051      ESCI_A.CR2.B.TXDMA = 1;         //Activate TX DMA Channel
00052      
00053      ESCI_A.SR.R = 0x80000000;           /* Clear TDRE flag */
00054    //  ESCI_A.DR.B.D = 'A';                      //Initial character
00055      
00056 /*  Configure eSCI B                                                              */
00057 /*  Configure per PLX Devices Interface- 19200, 8 Data Bits, 1 Stop Bit, No parity*/
00058 /*  Note:  TX and Rx Pads Configured in SIU_OPS.c                                 */
00059 
00060      ESCI_B.CR2.B.BSTP = 1;          //Suppress DMA TX with errors present
00061      ESCI_B.CR2.B.RXDMA = 0;         //Do Not Activate RX DMA Channel
00062      ESCI_B.CR2.B.TXDMA = 0;         //Do Not Activate TX DMA Channel, Yet
00063 
00064      ESCI_B.CR1.B.SBR = CPU_CLOCK/(16*BAUD_RATE);     //SCI Baud Rate
00065      ESCI_B.CR1.B.M = 0;             //1 start bit, 8 bits, 1 stop bit
00066      ESCI_B.CR1.B.PE = 0;            //Parity Disable
00067      ESCI_B.CR1.B.PT = 0;            //Parity Type
00068      ESCI_B.CR1.B.TE = 1;            //Transmitter Enable
00069      ESCI_B.CR1.B.RE = 1;            //Receiver Enable
00070 }
