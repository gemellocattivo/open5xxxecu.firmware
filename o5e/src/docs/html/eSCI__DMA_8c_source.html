00111         EDMA.TCD[18].CITER = EDMA.TCD[18].BITER = count;        // Inner 'Minor' Byte Count
00112         EDMA.TCD[18].SADDR = (vuint32_t) & SCI_TxQ0;    // reset Start Address
00113         EDMA.TCD[18].SLAST = -count;    // After major loop, reset src addr
00114 
00115         ESCI_A.SR.R = 0x80000000;       /* Clear TDRE flag */
00116 
00117         EDMA.SERQR.R = 18;      /* start DMA on this channel */
00118         EDMA.SSBR.R = 18;       /* Set channel 0 START bit to initiate first minor loop transfer */
00119 
00120         return count;           // how many we wrote    
00121 
00122 }                               // write_serial()
