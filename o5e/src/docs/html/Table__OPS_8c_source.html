00094 /* A, B, C, and D are the values bracketing the (MAP_value,speed_rpm) point                 */
00095   A = Table [MAP_index][RPM_index];
00096   B = Table[MAP_index+1][RPM_index];
00097   C = Table[MAP_index][RPM_index+1];
00098   D = Table[MAP_index+1][RPM_index+1];
00099 
00100 /* Perform Interpolation                                                                    */
00101   Intrm_Calc1 = ( ( ( B-A )*MAP_diff)>>4 ) +  A;
00102   Intrm_Calc2 = ( ( ( D-C )*MAP_diff)>>4)  +  C;
00103   Value = ( ( (Intrm_Calc2-Intrm_Calc1)*RPM_diff)>>3)  +  Intrm_Calc1;
